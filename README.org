Let's install xkblayout-state
#+BEGIN_SRC bash :export code
source_directory="$(cd $(dirname ${BASH_SOURCE[0]}) && pwd)"
git clone https://github.com/nonpop/xkblayout-state.git
cd xkblayout-state
make -j$(nproc)
cp xkblayout-state ${source_directory}/
#+END_SRC
